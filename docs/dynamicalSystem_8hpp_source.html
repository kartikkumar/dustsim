83  State operator( )( const Real time, const State& state )
84  {
85  const State position( { state[ astro::xPositionIndex ],
86  state[ astro::yPositionIndex ],
87  state[ astro::zPositionIndex ] } );
88 
89  // Compute the total acceleration acting on the system as a sum of the forces.
90  // Central body gravity is included by default.
91  State acceleration( astro::computeCentralBodyAcceleration( gravitationalParameter,
92  position ) );
93 
94  // Add J2 acceleration if model is set to active.
95  if ( isJ2AccelerationModelActive )
96  {
97  acceleration = acceleration + astro::computeJ2Acceleration( gravitationalParameter,
98  position,
99  equatorialRadius,
100  j2Coefficient );
101  }
102 
103  // Add solar radiation pressure acceleration if model is set to active.
104  if ( isRadiationPressureAccelerationModelActive )
105  {
106  // Compute unit vector to the Sun based on elapsed time.
107  // Inclination of central body around Sun is assumed to be zero.
108  const Real elapsedOrbitAngle = -solarMeanMotion * time;
109  const Vector unitVectorToSun(
110  { -std::cos( elapsedOrbitAngle ),
111  -std::cos( obliquity ) * std::sin( elapsedOrbitAngle ),
112  -std::sin( obliquity ) * std::sin( elapsedOrbitAngle ) } );
113 
114  acceleration = acceleration
115  + astro::computeCannonballRadiationPressureAcceleration(
116  radiationPressure,
117  radiationPressureCoefficient,
118  unitVectorToSun,
119  particleRadius,
120  particleBulkDensity );
121  }
122 
123  return State( { state[ 3 ],
